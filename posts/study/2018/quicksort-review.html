quicksort :: (Ord a) => [a] -> [a]
quicksort [] = []
quicksort (x:xs) =
  lessthan ++ [x] ++ largerthan
  where lessthan = quicksort [a | a <- xs, a < x]
        largerthan = quicksort [a | a <- xs, a >= x]
